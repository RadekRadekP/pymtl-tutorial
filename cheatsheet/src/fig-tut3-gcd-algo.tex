  \begin{Verbatim}[xleftmargin=0.5in]
def gcd( a, b ):
  while True:
    if a < b:
      a,b = b,a
    elif b != 0:
      a = a - b
    else:
      return a
  \end{Verbatim}

  \caption{\textbf{Euclid's GCD Algorithm --} Iteratively
    subtract the smaller value from the larger value until one of them is
    zero, at which time the GCD is the non-zero value. This is executable
    Python code.}
  \label{fig-tut3-gcd-algo}
